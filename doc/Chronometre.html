def initialize(chrono_label)
  @temps = 0
  @chrono_label = chrono_label
  @running = false

  super() do
    while @running
      update
      sleep(1)
    end
  end
end

def second_beautiful
  heures = @temps / 3600
  minutes = (@temps - 3600 * heures) / 60
  secondes = (@temps - 3600 * heures) - 60 * minutes

  "#{heures} : #{minutes} : #{secondes}"
end
