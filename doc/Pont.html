def afficher
  @tableau_case_pont.each do |case_pont|
    # Debug
    puts 'modif affichage case'

    case_pont.afficher_corp_pont_simple(@orientation) unless @double
    case_pont.afficher_corp_pont_double(@orientation) if @double
  end
end
